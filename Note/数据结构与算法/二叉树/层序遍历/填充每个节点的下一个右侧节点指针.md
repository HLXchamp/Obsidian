class Solution {
    public Node connect(Node root) {
        Queue<Node> queue = new ArrayDeque<>();
        int size = 0;
        if(root == null){
            return root;
        }
        queue.offer(root);
        Node node = root;
        Node prev = null;
        while(!queue.isEmpty()){
            size = queue.size();
            while(size > 0){
                node = queue.poll();
                if (prev != null) {
                    prev.next = node;
                }
                prev = node;
                if(node.left != null){
                    queue.offer(node.left);
                }
                if(node.right != null){
                    queue.offer(node.right);
                }
                size--;
            }                
            prev.next = null;
            prev = null;
        }
        return root;
    }
}
