新建一个变量存储每层最右节点的值，其他都不用变!

```java
class Solution {
    public List<Integer> rightSideView(TreeNode root) {
        Queue<TreeNode> queue = new ArrayDeque<>();
        int size = 0;
        List<Integer> res = new ArrayList<>();
        if(root == null){
            return res;
        }
        else{
            queue.offer(root);
        }
        TreeNode node = root;
        int right = 0;
        while(!queue.isEmpty()){
            size = queue.size();
            while(size > 0){
                node = queue.poll();
                right = node.val; //新建一个变量存储每层最右节点的值，其他都不用变
                if(node.left != null){
                    queue.offer(node.left);
                }
                if(node.right != null){
                    queue.offer(node.right);
                }
                size--;
            } 
            res.add(right);
        }
        return res;
    }
}
```
