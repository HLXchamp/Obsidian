和[[对称二叉树]]思路一模一样，只不过这个是比较左孩子与左孩子，右孩子与右孩子。

```java
class Solution {
    public boolean isSameTree(TreeNode p, TreeNode q) {
        if(p != null && q == null){
            return false;
        }
        else if(p == null && q != null){
            return false;
        }
        else if(p == null && q == null){
            return true;
        }
        else if(p.val != q.val){
            return false;
        }
        else{
            return isSameTree(p.left, q.left) && isSameTree(p.right, q.right);
        }
    }
}
```
